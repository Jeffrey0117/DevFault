# 🧠 核心邏輯流程

對每個 repo：

```
if (資料夾存在 && 有 .git)
    git pull
else
    git clone

if 有 branch
    git checkout

if 有 postInstall
    執行指令
```

---

# 🛠 devup.js（最簡原型）

```js
import { execSync } from "child_process"
import fs from "fs"
import path from "path"
import os from "os"

const config = JSON.parse(fs.readFileSync("./dev.config.json"))

const baseDir = config.baseDir.replace("~", os.homedir())

if (!fs.existsSync(baseDir)) {
  fs.mkdirSync(baseDir, { recursive: true })
}

for (const repo of config.repos) {
  const target = path.join(baseDir, repo.name)

  if (fs.existsSync(path.join(target, ".git"))) {
    console.log(`🔄 Pulling ${repo.name}`)
    execSync(`git -C ${target} pull`, { stdio: "inherit" })
  } else {
    console.log(`📥 Cloning ${repo.name}`)
    execSync(`git clone ${repo.url} ${target}`, { stdio: "inherit" })
  }

  if (repo.branch) {
    execSync(`git -C ${target} checkout ${repo.branch}`, { stdio: "inherit" })
  }

  if (repo.postInstall) {
    execSync(`cd ${target} && ${repo.postInstall}`, { stdio: "inherit" })
  }
}
```
